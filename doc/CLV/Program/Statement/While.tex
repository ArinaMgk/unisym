\textbf{EXAMPLE for }\textbf{\textit{Python}}

\begin{lstlisting}[language=Python]
	numbers = [0, 6, 1, 8, 2, 8]
	even = []
	odd = []
	while len(numbers) > 0 :
		number = numbers.pop()
		if(number % 2 == 0):
			even.append(number)
		else:
			odd.append(number)
\end{lstlisting}
